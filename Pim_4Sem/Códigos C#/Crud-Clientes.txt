        private void btnSalvar_Click(object sender, EventArgs e)
        {
            //strSQL = "Insert Into Pim (Nome,Conta_Banco,Email,Senha,DataNasc) values (@Nome,@Conta_Banco,@Email,@Senha,@DataNasc)";

            conn = new SqlConnection(strconn);

            SqlCommand cmd = new SqlCommand(strSQL,conn);
            cmd.CommandText = "Insert Into tbUser (idUser,Nome,Conta_Banco,Email,Senha,DataNasc,idSaldo) values (@idUser,@Nome,@Conta_Banco,@Email,@Senha,@DataNasc,@idSaldo)";
            cmd.Parameters.Add("@idSaldo", SqlDbType.VarChar).Value = txtIdSaldo.Text;
            cmd.Parameters.Add("@idUser", SqlDbType.VarChar).Value = txtidUser.Text;
            cmd.Parameters.Add("@Nome",SqlDbType.VarChar).Value = txtNome.Text;
            cmd.Parameters.Add("@Conta_Banco", SqlDbType.VarChar).Value = txtConta.Text;
            cmd.Parameters.Add("@Email", SqlDbType.VarChar).Value = txtEmail.Text;
            cmd.Parameters.Add("@Senha", SqlDbType.VarChar).Value = txtSenha.Text;
            cmd.Parameters.Add("@DataNasc", SqlDbType.Date).Value = maskedtxtDataNasc.Text;

            cmd.CommandText = "Insert Into tbSaldo(idSaldo,Saldo_R$,Saldo_bitcoin) values(@idS,@R$,@bit)";
            cmd.Parameters.Add("@idS", SqlDbType.VarChar).Value = txtIdSaldo.Text;
            cmd.Parameters.Add("@R$", SqlDbType.Float).Value = txtReais.Text;
            cmd.Parameters.Add("@bit", SqlDbType.Float).Value = txtBitCoin.Text;
            try
            {
                conn.Open();
                cmd.ExecuteNonQuery();
                MessageBox.Show("Cadastrado com sucesso!");
            }
            catch (SqlException ex)
            {
                MessageBox.Show(ex.Message);
            }
            finally
            {
                conn.Close();
            }
            txtidUser.Clear();
            txtIdSaldo.Clear();
            txtConta.Clear();
            txtEmail.Clear();
            txtNome.Clear();
            txtSenha.Clear();
            maskedtxtDataNasc.Clear();
            txtReais.Clear();
            txtBitCoin.Clear();
            dataGridView1.Refresh();
            dataGridView2.Refresh();
        }
